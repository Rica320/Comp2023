import io;
class Simple {

    int a;
    public Simple newS() {
        return this;
    }

    public int getA() {
        return a;
    }

    public Simple setA(int e) {
        a = e;
        return this;
    }

    public static void main(String[] args) {
        io.println(new Simple().setA(3).newS().getA());
    }
}
