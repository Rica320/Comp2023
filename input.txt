import io;

class AssignInnerScope {

    int a() {
        int[] a;
        a = new int[10];
        a[0] = 1;
        return a[0];
    }

    public static void main(String[] args) {

        int[] a;
        int[] b;
        int[] c;

        a = new int[10];
        b = new int[10];
        c = new int[10];

        a[0] = 1;
        b[0] = 0;
        c[0] = 3;

        a[b[0]] = 5;

        io.println(a[0]);
    }
}
