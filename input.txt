import io;

class AssignInnerScope {

int[] b;

    int[] retint(int[] d) {

                b= new int[10];
                b[d[0]] = 1;

        return b;
    }

    public static void main(String[] args) {

        int[] a;
        a= new int[10];
        a[0] = 0;


        io.println(new AssignInnerScope().retint(a)[0]);
        io.println(a.length);

    }
}
